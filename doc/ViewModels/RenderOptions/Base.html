def initialize prefix, name, options

  self.template_name = deoptionize name
  @format = @options.delete :format
end

def to_render_options
  @options[:locals] ||= {}
  @options[:locals].reverse_merge! :view_model => view_model
  @options.reverse_merge :file => file
end
